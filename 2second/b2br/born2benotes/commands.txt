reboot:
sudo reboot

login as root:
su -

check if root:
whoami

view partitions:
lsblk

check if UFW is started:
sudo ufw status

check if SSH is started:
sudo systemctl status ssh

connect to ssh from host
ssh your_username@127.0.0.1 -p 4242

restart ssh service:
sudo systemctl restart ssh

check if group created:
getent group group_name

create new user:
sudo adduser new_username

delete user:
sudo userdel username

create new group:
sudo groupadd user42
sudo groupadd evaluating

add user to groups:
sudo usermod -aG user42 username
sudo usermod -aG evaluating username

delete group:
sudo groupdel groupname 

open password rules file:
sudo nano /etc/pam.d/common-password

open password expiration file:
sudo nano /etc/login.defs

check hostname:
hostnamectl

change hostname:
hostnamectl set-hostname new_hostname
sudo nano /etc/hosts
...
127.0.0.1       localhost
127.0.0.1       new_hostname
...
sudo reboot

add new rule to open port:
sudo ufw allow 8080

delete port:
sudo ufw status numbered
sudo ufw delete number

config cron
sudo crontab -u root -e

check cron jobs:
sudo crontab -u root -l
